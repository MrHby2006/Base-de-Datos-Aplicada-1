--Creació de tablas

CREATE TABLE cliente(
run varchar2(15) primary key,
nombre varchar2(50) not null,
direccion varchar2(60) not null,
telefono varchar2(10)
);

CREATE TABLE vehiculo(
patente varchar2(8) primary key,
marca varchar2(40) not null,
modelo varchar2(50) not null,
run varchar2(15), foreign key(run) references cliente(run) -- Clave foranea de cliente (parece sacado de los modelos relacionales)
);
